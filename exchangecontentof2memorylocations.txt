data segment
    MSG1 DB 10,13,'ENTER FIRST VALUE: $'
    MSG2 DB 10,13,'ENTER SECOND VALUE: $'
    MSG3 DB 10,13,'AFTER EXCHANGE:$'
    A DB ?
    B DB ?
data ends

code segment
    ASSUME CS:code DS:data

START:
    MOV AX,data
    MOV DS,AX
    LEA DX,MSG1
    MOV AH,09H
    INT 21H
    MOV AH,01H
    INT 21H
    MOV A,AL
    LEA DX,MSG2
    MOV AH,09H
    INT 21H
    MOV AH,01H
    INT 21H
    MOV B,AL
    MOV AL,A
    XCHG AL,B
    MOV A,AL
    LEA DX,MSG3
    MOV AH,09H
    INT 21H
    MOV DL,A
    MOV AH,02H
    INT 21H
    MOV DL,B
    MOV AH,02H
    INT 21H
    MOV AH,4CH
    INT 21H
code ends
end start
