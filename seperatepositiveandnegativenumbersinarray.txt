data segment
    MSG1 DB 10,13,'ENTER COUNT: $'
    MSG2 DB 10,13,'ENTER NUMBERS: $'
    MSG3 DB 10,13,'POSITIVE NUMBERS: $'
    MSG4 DB 10,13,'NEGATIVE NUMBERS: $'
    ARR DB 20 DUP(?)
data ends

code segment
    ASSUME CS:code DS:data

START:
    MOV AX,data
    MOV DS,AX
    LEA DX,MSG1
    MOV AH,09H
    INT 21H
    MOV AH,01H
    INT 21H
    SUB AL,30H
    MOV CL,AL
    LEA DX,MSG2
    MOV AH,09H
    INT 21H
    LEA SI,ARR
INPUT:
    MOV AH,01H
    INT 21H
    CMP AL,0DH
    JE DONE
    SUB AL,30H
    MOV [SI],AL
    INC SI
    LOOP INPUT
DONE:
    LEA DX,MSG3
    MOV AH,09H
    INT 21H
    LEA SI,ARR
    MOV CL,AL
    MOV CH,0
    MOV BL,0
PRINT_POS:
    MOV AL,[SI]
    CMP AL,0
    JL NEXT
    ADD AL,30H
    MOV DL,AL
    MOV AH,02H
    INT 21H
NEXT:
    INC SI
    LOOP PRINT_POS
    LEA DX,MSG4
    MOV AH,09H
    INT 21H
    LEA SI,ARR
    MOV CL,AL
    MOV CH,0
PRINT_NEG:
    MOV AL,[SI]
    CMP AL,0
    JGE SKIP
    NEG AL
    ADD AL,30H
    MOV DL,'-'
    MOV AH,02H
    INT 21H
    MOV DL,AL
    MOV AH,02H
    INT 21H
SKIP:
    INC SI
    LOOP PRINT_NEG
    MOV AH,4CH
    INT 21H
code ends
end start
