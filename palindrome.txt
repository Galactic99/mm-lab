data segment
       MSG1 DB 10,13,'ENTER THE STRING: $'
       MSG2 DB 10,13,'STRING IS PALINDROME$'
       MSG3 DB 10,13,'STRING IS NOT PALINDROME$'
       STRING DB 50 DUP(?)
data ends

code segment
    ASSUME CS:code DS:data
    
    START:
        MOV AX,data
        MOV DS,AX
        LEA DX,MSG1
        MOV AH,09H
        INT 21H
        LEA SI,STRING
        LEA DI,STRING
        MOV AH,01H
    READ:
        INT 21H 
        CMP AL,0DH
        JE TERMINATE
        MOV [DI],AL 
        INC DI
        JMP READ
    TERMINATE:
        MOV AL,'$'
        MOV [DI],AL
    DOTHIS:
        DEC DI
        MOV AL,[SI]
        CMP [DI],AL
        JNE NOTPALINDROME
        INC SI
        CMP SI,DI
        JL DOTHIS
    PALINDROME:
        LEA DX,MSG2
        MOV AH,09H
        INT 21H
        JMP XX
    NOTPALINDROME:
        LEA DX,MSG3
        MOV AH,09H
        INT 21H
    XX:
        MOV AH,4CH
        INT 21H
code ends
end start
