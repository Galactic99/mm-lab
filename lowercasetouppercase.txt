data segment
    MSG1 DB 10,13,'ENTER THE STRING: $'
    MSG2 DB 10,13,'UPPERCASE STRING: $'
    STRING DB 50 DUP(?)
data ends

code segment
    ASSUME CS:code DS:data

START:
    MOV AX,data
    MOV DS,AX
    LEA DX,MSG1
    MOV AH,09H
    INT 21H
    LEA DI,STRING
    MOV AH,01H
READ:
    INT 21H
    CMP AL,0DH
    JE TERM
    MOV [DI],AL
    INC DI
    JMP READ
TERM:
    MOV AL,'$'
    MOV [DI],AL
    LEA SI,STRING
CONVERT:
    MOV AL,[SI]
    CMP AL,'$'
    JE DISPLAY
    CMP AL,'a'
    JB SKIP
    CMP AL,'z'
    JA SKIP
    SUB AL,20H
    MOV [SI],AL
SKIP:
    INC SI
    JMP CONVERT
DISPLAY:
    LEA DX,MSG2
    MOV AH,09H
    INT 21H
    LEA DX,STRING
    MOV AH,09H
    INT 21H
    MOV AH,4CH
    INT 21H
code ends
end start
