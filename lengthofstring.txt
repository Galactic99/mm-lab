data segment
    MSG1 DB 10,13,'ENTER THE STRING: $'
    MSG2 DB 10,13,'LENGTH OF STRING: $'
    STRING DB 50 DUP(?)
data ends

code segment
    ASSUME CS:code DS:data

START:
    MOV AX,data
    MOV DS,AX
    LEA DX,MSG1
    MOV AH,09H
    INT 21H
    LEA DI,STRING
    MOV AH,01H
READ:
    INT 21H
    CMP AL,0DH
    JE TERMINATE
    MOV [DI],AL
    INC DI
    JMP READ
TERMINATE:
    MOV AL,'$'
    MOV [DI],AL
    LEA SI,STRING
    MOV CL,0
COUNT:
    MOV AL,[SI]
    CMP AL,'$'
    JE DISPLAY
    INC CL
    INC SI
    JMP COUNT
DISPLAY:
    LEA DX,MSG2
    MOV AH,09H
    INT 21H
    ADD CL,30H
    MOV DL,CL
    MOV AH,02H
    INT 21H
    MOV AH,4CH
    INT 21H
code ends
end start
