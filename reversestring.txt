data segment
    MSG1 DB 10,13,'ENTER THE STRING: $'
    MSG2 DB 10,13,'REVERSED STRING: $'
    STRING DB 50 DUP(?)
data ends

code segment
    ASSUME CS:code DS:data

START:
    MOV AX,data
    MOV DS,AX
    LEA DX,MSG1
    MOV AH,09H
    INT 21H
    LEA DI,STRING
    MOV AH,01H
READ:
    INT 21H
    CMP AL,0DH
    JE TERMINATE
    MOV [DI],AL
    INC DI
    JMP READ
TERMINATE:
    MOV AL,'$'
    MOV [DI],AL
    DEC DI
    LEA DX,MSG2
    MOV AH,09H
    INT 21H
REVERSE:
    MOV DL,[DI]
    CMP DL,'$'
    JE EXIT
    MOV AH,02H
    INT 21H
    DEC DI
    CMP DI,OFFSET STRING
    JGE REVERSE
EXIT:
    MOV AH,4CH
    INT 21H
code ends
end start
