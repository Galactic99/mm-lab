data segment
    MSG1 DB 10,13,'ENTER THE STRING: $'
    MSG2 DB 10,13,'NUMBER OF VOWELS: $'
    STRING DB 50 DUP(?)
data ends

code segment
    ASSUME CS:code DS:data

START:
    MOV AX,data
    MOV DS,AX
    LEA DX,MSG1
    MOV AH,09H
    INT 21H
    LEA DI,STRING
    MOV AH,01H
READ:
    INT 21H
    CMP AL,0DH
    JE TERM
    MOV [DI],AL
    INC DI
    JMP READ
TERM:
    MOV AL,'$'
    MOV [DI],AL
    MOV CL,0
    LEA SI,STRING
NEXT:
    MOV AL,[SI]
    CMP AL,'$'
    JE DISPLAY
    CMP AL,'A'
    JE VOW
    CMP AL,'E'
    JE VOW
    CMP AL,'I'
    JE VOW
    CMP AL,'O'
    JE VOW
    CMP AL,'U'
    JE VOW
    CMP AL,'a'
    JE VOW
    CMP AL,'e'
    JE VOW
    CMP AL,'i'
    JE VOW
    CMP AL,'o'
    JE VOW
    CMP AL,'u'
    JE VOW
    JMP SKIP
VOW:
    INC CL
SKIP:
    INC SI
    JMP NEXT
DISPLAY:
    LEA DX,MSG2
    MOV AH,09H
    INT 21H
    ADD CL,30H
    MOV DL,CL
    MOV AH,02H
    INT 21H
    MOV AH,4CH
    INT 21H
code ends
end start
